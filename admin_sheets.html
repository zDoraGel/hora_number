async function doBackup() { const month = parseInt(document.getElementById('bkMonth').value); const year = parseInt(document.getElementById('bkYear').value); const btn = document.getElementById('bkBtn'); const res = document.getElementById('bkResult'); const label = month === 0 ? `ทั้งปี พ.ศ. ${year+543}` : `${MONTHS[month]} พ.ศ. ${year+543}`; btn.disabled = true; btn.innerHTML = 'กำลัง Backup...'; res.innerHTML = ''; try { const data = await api({ action:'backup_db', token:TOKEN, month, year }); if (!data.ok) { res.innerHTML = `
❌ ${data.msg}
`; toast('❌ Backup ไม่สำเร็จ'); } else { res.innerHTML = `
✅ Backup สำเร็จ!
📅 ช่วงเวลา: ${label}
📄 ไฟล์: ${data.filename}
📦 ขนาด: ${data.size_kb} KB
`; toast(`✅ Backup ${label} สำเร็จ!`); loadBackups(); } } catch(e) { res.innerHTML = `
❌ เชื่อมต่อไม่ได้ — ตรวจสอบ SCRIPT_URL
`; toast('❌ เชื่อมต่อล้มเหลว'); } btn.disabled = false; btn.innerHTML = '💾 สร้าง Backup'; } async function loadBackups() { const out = document.getElementById('bkList'); out.innerHTML = '
กำลังโหลด...
'; try { const data = await api({ action:'list_backups', token:TOKEN }); if (!data.ok) { out.innerHTML='

'+data.msg+'

'; return; } const list = data.backups || []; if (!list.length) { out.innerHTML = '

ยังไม่มี Backup ค่ะ

'; return; } out.innerHTML = `

ทั้งหมด ${list.length} ไฟล์

${list.map(b => ` `).join('')}
ไฟล์ ช่วงเวลา ขนาด สร้างเมื่อ
${b.filename} ${b.label||'-'} ${b.filesize ? Math.round(b.filesize/1024)+' KB' : '-'} ${fmtShort(b.created_at)}
`; } catch(e) { out.innerHTML='

เชื่อมต่อล้มเหลว

'; } } function downloadBackup(filename) { const url = SCRIPT_URL + '?action=download_backup' + '&file=' + encodeURIComponent(filename) + '&token=' + encodeURIComponent(TOKEN); const a = document.createElement('a'); a.href = url; a.download = filename; a.click(); } async function deleteBackup(filename) { if (!confirm(`ลบไฟล์ Backup\n"${filename}"\nแน่ใจไหมคะ?`)) return; try { const data = await api({ action:'delete_backup', token:TOKEN, file:filename }); if (data.ok) { toast('🗑️ ลบแล้วค่ะ'); loadBackups(); } else toast('❌ ' + data.msg); } catch(e) { toast('❌ เชื่อมต่อล้มเหลว'); } }
